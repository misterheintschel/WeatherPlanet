CREATE TABLE public.users (
	id serial4 NOT NULL,
	namef varchar(30) NULL,
	namel varchar(30) NULL,
	email varchar(30) NULL,
	"password" text NULL,
	CONSTRAINT users_pkey PRIMARY KEY (id),
	CONSTRAINT users_un UNIQUE (email)
);
